is.binary <- function(x) { # function to check whether or not a variable is binary 
  n.unique <- length(unique(x))
  min <- min(x)
  max <- max(x)
  if (n.unique == 2 & min == 0 & max == 1) {
    return(TRUE)
  }
  else {
    return(FALSE)
  }
}
